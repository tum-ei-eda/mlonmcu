[docs]def get_template_text(name):
    return pkgutil.get_data(
        "mlonmcu", os.path.join("..", "templates", name + ".yml.j2")
    )




[docs]def fill_template(name, data={}):
    template_text = get_template_text(name)
    if template_text:
        text = template_text.decode("utf-8")
        tmpl = jinja2.Template(text)
        rendered = tmpl.render(**data)
        return rendered
    return None




[docs]def fill_environment_yaml(template_name, home_dir):
